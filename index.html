    JavaScript:

    					
var testJson = {
	limit : 999,
	minLength : 0,
	type : "number"
}
function changeInputType(item){
	var index = item.selectedIndex ;
		switch(Number(item.options[index].value)){
			case 1 :
				testJson.type = "number";
			break;
			case 2 : 
				testJson.charWord = false;
				testJson.type = "string";
			break;
			case 3:
				testJson.charWord = true;
				testJson.type = "string";
			break;
		}
}
function strJudge( item ){
	testJson.val = item.value;
	var json = $tools.strJudge(testJson);
	item.value = json.val
}

    	params : {
		parent : obj,		// dom元素  用来作为table容器的
		config : [
			{
				name: String,		//表名称
				row:Number,			//每页显示多少行	默认10条
				total:Number,		//数据总数量
				show:Number,		//可视的数据条数
				check: Event:function( result ){		//事件,若写有该函数，就会自动添加复选框，当点击的复选框的时候就会进行触发，点击全部，返回的当前页的所有数据的信息，点击单个就会返回当前的数据
					console.log(result)
				},
				serial:Boolean,		// 是否显示序列号，默认不显示
				pagingCallBack:Event:function(pageNum,callBack){
					// 返回当前的页码
					console.log(pageNum);
					// 将得到的数据重新填冲
					callBack(data);
					
				},
				fields:Array:[
					{
						name: String,	//字段名称
						key: String,	//索引值
						style: JSON:{	//给该字段给予自定义样式
							width: String,	 
						},
						methods: JSON {		//方法集
							onChange: Event:function(result){	//当绑定了该函数后,在该字段的数据里面就会有编辑的小按钮，如果改变数据的时候就会触发
								
								console.log(result);
							}
						},
						template:String | Event: function( value ){		//当绑定了该字段后,就可以自定该数据框的内容.可以是字符串，也可以是函数，函数的参数是当前的字段数据
							
							console.log(result)
						}
					}
				]
			}
		]
